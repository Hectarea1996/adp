#### ***Macro*** ADP-PRIVATE:WITH-DEFCLASS-COMPONENTS

```Lisp
(DEFMACRO ADP-PRIVATE:WITH-DEFCLASS-COMPONENTS
          (((&REST #:COMPONENT-REST-ARGS0) #:FUNCTION-BODY-ARG1)
           &BODY #:BODY-ARG2)
  (ADP-PRIVATE::CHECK-COMPONENT-SYMBOLS #:COMPONENT-REST-ARGS0
                                        '(CLASS-NAME
                                          ADP-PRIVATE::SUPERCLASS-NAMES
                                          ADP-PRIVATE::SLOT-SPECIFIERS
                                          ADP-PRIVATE::SLOT-NAMES
                                          ADP-PRIVATE::SLOT-OPTIONS
                                          ADP-PRIVATE::READER-FUNCTION-NAMES
                                          ADP-PRIVATE::WRITER-FUNCTION-NAMES
                                          ADP-PRIVATE::ACCESSOR-FUNCTION-NAMES
                                          ADP-PRIVATE::ALLOCATION-TYPES
                                          ADP-PRIVATE::INITARG-NAMES
                                          ADP-PRIVATE::INITFORMS
                                          ADP-PRIVATE::TYPE-SPECIFIERS
                                          ADP-PRIVATE::SLOT-DOCUMENTATIONS
                                          ADP-PRIVATE::CLASS-OPTIONS
                                          ADP-PRIVATE::DEFAULT-INITARGS
                                          DOCUMENTATION
                                          ADP-PRIVATE::METACLASS))
  (LET* ((#:FUNCTION-BODY-VALUE-VAR3 (GENSYM))
         (#:LET-BINDINGS-VAR4
          (MAPCAR
           (LAMBDA (#:LAMBDA-VAR5)
             (LIST #:LAMBDA-VAR5
                   (LIST
                    (FIND-SYMBOL
                     (CONCATENATE 'STRING "DEFCLASS" "-"
                                  (SYMBOL-NAME #:LAMBDA-VAR5))
                     :ADPPVT)
                    #:FUNCTION-BODY-VALUE-VAR3)))
           #:COMPONENT-REST-ARGS0)))
    `(LET ((,#:FUNCTION-BODY-VALUE-VAR3 ,#:FUNCTION-BODY-ARG1))
       (LET ,#:LET-BINDINGS-VAR4
         ,@#:BODY-ARG2))))
```

(DEFUN ADP-PRIVATE:DEFINE-COMPILER-MACRO-DECLARATIONS
       (ADP-PRIVATE::DEFINE-COMPILER-MACRO-SOURCE)
  (LET ((ADP-PRIVATE::POST-LAMBDA-LIST
         (CDDDR ADP-PRIVATE::DEFINE-COMPILER-MACRO-SOURCE)))
    (LOOP ADP-PRIVATE::FOR ADP-PRIVATE::EXPR ADP-PRIVATE::IN ADP-PRIVATE::POST-LAMBDA-LIST
          ADP-PRIVATE::WHILE (OR (ADP-PRIVATE::DECLARATIONP ADP-PRIVATE::EXPR)
                                 (ADP-PRIVATE::DOCUMENTATIONP
                                  ADP-PRIVATE::EXPR))
          WHEN (ADP-PRIVATE::DECLARATIONP ADP-PRIVATE::EXPR)
          ADP-PRIVATE::COLLECT ADP-PRIVATE::EXPR)))

                                 (ADP-PRIVATE::DOCUMENTATIONP
